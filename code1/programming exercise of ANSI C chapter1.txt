//3)print a pattern

{

    while(i<=4)
    {
        while(i>=j)
        {
            printf("* ");
              j=j+1;
        }
        j=1;
        i=i+1;
        printf("\n");
    }

}

{

   float s;

//12)given center and a point on circumference find radius,perimeter and area

#include<math.h>
#include<conio.h>

{
   int x1=0,x2=4,y1=0,y2=5;
   float p=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
   printf("radius:%f\n",p);
   printf("perimeter:%f\n",2*3.14*p);
   printf("area:%f\n",3.14*p*p);
